            printf("\nTotal value : %d",total);
            pro_stat=total-sale[id-1];
            printf("\n Product status after increment %d",pro_stat);

        case 3:
        {
            printf("\nEnter the product id [1,2,3]: ");
            scanf("%d",&id);
            int x;
            x=total-sale[id-1];
            if(x<0)
            printf("\nBy increasing sale of %-d we get break even",x);
            else if(x>0)
            printf("\nBy decreasing sale of %d we get break even",x);

        case 4:
        {
            printf("\nExit");
        }
        default:

        }while(key!=4);
    }

}
